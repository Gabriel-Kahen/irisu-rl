function(irisu_enable_warnings target)
  target_compile_options(${target} PRIVATE
    $<$<CXX_COMPILER_ID:GNU,Clang>:-Wall;-Wextra;-Wpedantic;-Wconversion;-Wshadow>
  )
endfunction()

if(IRISU_PHYSICS_BACKEND STREQUAL "portable")
  add_subdirectory(third_party/box2d_legacy)
  set_target_properties(irisu_box2d_legacy PROPERTIES
    CXX_VISIBILITY_PRESET hidden
    POSITION_INDEPENDENT_CODE ON
    VISIBILITY_INLINES_HIDDEN YES
  )
  set(irisu_physics_source clone/core/physics.cpp)
  set(irisu_physics_target irisu::box2d_legacy)
  set(irisu_physics_backend_label "portable-gnu-r58")
elseif(IRISU_PHYSICS_BACKEND STREQUAL "exact-msvc")
  if(NOT CMAKE_SIZEOF_VOID_P EQUAL 4)
    message(FATAL_ERROR
      "IRISU_PHYSICS_BACKEND=exact-msvc requires a 32-bit build (configure with -m32)"
    )
  endif()
  if(NOT CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
    message(FATAL_ERROR
      "The exact MSVC host currently requires a 32-bit GNU C++ consumer"
    )
  endif()
  if(IRISU_EXACT_BOX2D_LIBRARY STREQUAL "" OR
     NOT EXISTS "${IRISU_EXACT_BOX2D_LIBRARY}")
    message(FATAL_ERROR
      "Set IRISU_EXACT_BOX2D_LIBRARY to the ELF32 output of "
      "tools/host-msvc9-box2d-multiworld.py"
    )
  endif()
  file(READ "${IRISU_EXACT_BOX2D_LIBRARY}" irisu_exact_elf_header
       OFFSET 0 LIMIT 5 HEX)
  if(NOT irisu_exact_elf_header STREQUAL "7f454c4601")
    message(FATAL_ERROR
      "IRISU_EXACT_BOX2D_LIBRARY must be a 32-bit ELF library"
    )
  endif()
  execute_process(
    COMMAND "${CMAKE_NM}" -D --defined-only -j
            "${IRISU_EXACT_BOX2D_LIBRARY}"
    RESULT_VARIABLE irisu_exact_nm_status
    OUTPUT_VARIABLE irisu_exact_exports
    ERROR_VARIABLE irisu_exact_nm_error
  )
  string(FIND "\n${irisu_exact_exports}\n" "\nb2d_world_create\n"
         irisu_exact_world_create_offset)
  if(NOT irisu_exact_nm_status EQUAL 0 OR
     irisu_exact_world_create_offset EQUAL -1)
    message(FATAL_ERROR
      "IRISU_EXACT_BOX2D_LIBRARY does not expose the required multiworld API: "
      "${irisu_exact_nm_error}"
    )
  endif()
  set(irisu_readelf "${CMAKE_READELF}")
  if(irisu_readelf STREQUAL "" OR NOT EXISTS "${irisu_readelf}")
    find_program(irisu_readelf NAMES readelf)
  endif()
  if(NOT irisu_readelf)
    message(FATAL_ERROR
      "Cannot validate IRISU_EXACT_BOX2D_LIBRARY: readelf was not found"
    )
  endif()
  execute_process(
    COMMAND "${CMAKE_COMMAND}" -E env "LC_ALL=C"
            "${irisu_readelf}" -d "${IRISU_EXACT_BOX2D_LIBRARY}"
    RESULT_VARIABLE irisu_exact_readelf_status
    OUTPUT_VARIABLE irisu_exact_dynamic_section
    ERROR_VARIABLE irisu_exact_readelf_error
  )
  set(irisu_exact_actual_soname "<missing>")
  if(irisu_exact_readelf_status EQUAL 0)
    string(REGEX MATCH "SONAME[^[]*\\[([^]]+)\\]"
           irisu_exact_soname_match "${irisu_exact_dynamic_section}")
    if(irisu_exact_soname_match)
      set(irisu_exact_actual_soname "${CMAKE_MATCH_1}")
    endif()
  endif()
  if(NOT irisu_exact_readelf_status EQUAL 0 OR
     NOT irisu_exact_actual_soname STREQUAL irisu_exact_box2d_soname)
    message(FATAL_ERROR
      "IRISU_EXACT_BOX2D_LIBRARY must carry DT_SONAME "
      "'${irisu_exact_box2d_soname}' (found '${irisu_exact_actual_soname}'). "
      "Regenerate it with tools/host-msvc9-box2d-multiworld.py; refusing "
      "to create an absolute or unstable DT_NEEDED entry. "
      "${irisu_exact_readelf_error}"
    )
  endif()
  execute_process(
    COMMAND "${CMAKE_COMMAND}" -E env "LC_ALL=C"
            "${irisu_readelf}" -rW "${IRISU_EXACT_BOX2D_LIBRARY}"
    RESULT_VARIABLE irisu_exact_relocations_status
    OUTPUT_VARIABLE irisu_exact_relocations
    ERROR_VARIABLE irisu_exact_relocations_error
  )
  if(NOT irisu_exact_relocations_status EQUAL 0)
    message(FATAL_ERROR
      "Cannot inspect exact library relocations: ${irisu_exact_relocations_error}"
    )
  endif()
  if(irisu_exact_relocations MATCHES
     "R_386_(JUMP_SLOT|GLOB_DAT)[^\n]*msvc_b2d_")
    message(FATAL_ERROR
      "IRISU_EXACT_BOX2D_LIBRARY contains preemptable internal msvc_b2d_* "
      "relocations; regenerate it with tools/host-msvc9-box2d-multiworld.py"
    )
  endif()
  file(SHA256 "${IRISU_EXACT_BOX2D_LIBRARY}" irisu_exact_box2d_sha256)
  add_library(irisu_box2d_msvc_exact SHARED IMPORTED GLOBAL)
  set_target_properties(irisu_box2d_msvc_exact PROPERTIES
    IMPORTED_LOCATION "${IRISU_EXACT_BOX2D_LIBRARY}"
    IMPORTED_SONAME "${irisu_exact_box2d_soname}"
  )
  set(irisu_physics_source
      tools/exact-physics-prototype/physics_wrapper_forward.cpp)
  set(irisu_physics_target irisu_box2d_msvc_exact)
  set(irisu_physics_backend_label "exact-msvc9-r58-multiworld-forward")
else()
  message(FATAL_ERROR
    "Unknown IRISU_PHYSICS_BACKEND=${IRISU_PHYSICS_BACKEND}; expected portable or exact-msvc"
  )
endif()

add_library(irisu_core STATIC
  clone/core/config.cpp
  clone/core/config_io.cpp
  clone/core/dx_random.cpp
  clone/core/normal_rules.cpp
  ${irisu_physics_source}
  clone/core/simulator.cpp
)
set_target_properties(irisu_core PROPERTIES
  CXX_VISIBILITY_PRESET hidden
  POSITION_INDEPENDENT_CODE ON
  VISIBILITY_INLINES_HIDDEN YES
)
target_include_directories(irisu_core PUBLIC clone/include)
if(IRISU_LEGACY_FP_MODE STREQUAL "x87")
  target_compile_definitions(irisu_core PUBLIC IRISU_X87_FIDELITY_MODE=1)
endif()
if(IRISU_GNU_X87_CONTROL_WORD_CAPABLE)
  target_compile_definitions(irisu_core PUBLIC
    IRISU_GNU_X87_CONTROL_WORD_ENVIRONMENT=1
  )
endif()
target_link_libraries(irisu_core PRIVATE ${irisu_physics_target})
if(NOT EMSCRIPTEN)
  target_link_libraries(irisu_core PUBLIC Threads::Threads)
endif()
if(IRISU_PHYSICS_BACKEND STREQUAL "exact-msvc")
  target_compile_definitions(irisu_core PRIVATE
    IRISU_EXACT_LIBRARY_SONAME="${irisu_exact_box2d_soname}"
  )
  target_link_libraries(irisu_core PRIVATE ${CMAKE_DL_LIBS})
  # The wrapper deliberately resolves a typed call table at runtime. Retain the
  # genuine exact DSO as DT_NEEDED so RTLD_NOLOAD cannot be redirected by search
  # path changes or by an unrelated preload interposer.
  target_link_options(irisu_core INTERFACE "LINKER:--no-as-needed")
endif()
irisu_enable_warnings(irisu_core)
